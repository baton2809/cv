Parametrized filtering of MONGO entries in RESTful API


@SuppressWarnings("unchecked")
private FilterOperationResult getFilterOperation(Map filter, String by) {
    List filterOperations = new ArrayList<>();
    var sortOperation = Aggregation.sort(Sort.by(Sort.Direction.DESC, ENTITY_ID));
    List searchOperations = new ArrayList<>();
    if (filter != null) {
        for (Map.Entry entry : filter.entrySet()) {
            var key = entry.getKey();
            var value = entry.getValue();

            if (key.equals(SORT)) {
                var sortCriteria = (List>) value;
                List orders = new ArrayList<>();
                for (Map criteria : sortCriteria) {
                    var field = by == null ? criteria.get(FIELD) : ID;
                    var order = criteria.get(DIRECTION);
                    var direction = Sort.Direction.fromString(order);
                    orders.add(new Sort.Order(direction, field));
                }
                sortOperation = Aggregation.sort(Sort.by(orders));
            } else if (key.equals(SEARCH)) {
                var groupByClauseEmpty = !StringUtils.hasText(by);

                if (!numeric(value.toString())) {
                    searchOperations.add(groupByClauseEmpty ?
                            null :
                            Aggregation.match(Criteria.where(by).regex(value.toString(), "i")));
                } else {
                    searchOperations.add(Aggregation.addFields()
                            .addField(by + STRING)
                            .withValue(ConvertOperators.ToString.toString("$" + by))
                            .build());
                    searchOperations.add(Aggregation.match(Criteria.where(by + STRING).regex(value.toString())));
                    searchOperations.add(Aggregation.project()
                            .andExclude(by + STRING));
                }
            } else {
                if (value instanceof List) {
                    filterOperations.add(Aggregation.match(Criteria.where(key).in((List) value)));
                } else {
                    var from = (String) ((LinkedHashMap) value).get(FROM);
                    var to = (String) ((LinkedHashMap) value).get(TO);

                    var dateCriteria = Criteria.where(key);

                    dateCriteriaHandler(filterOperations, from, to, dateCriteria);
                }
            }
        }
    }
    return new FilterOperationResult(filterOperations, sortOperation, searchOperations);
}

Usage of RegisterBeanDefinition in order to create the bean using key from YAML file


@Configuration
public class ApplicationConfig {

    private static final String ALGORITHM = "";

    @Value("${encryptor.pool-size}")
    private int encryptorPoolSize;
    @Value("${encryptor.enabled}")
    private boolean encryptionEnabled;

    @Autowired
    private KeysProperties keysProperties;
    @Autowired
    private GenericApplicationContext context;

    @PostConstruct
    public void init() {
        if (encryptionEnabled) {
            keysProperties.getKeys().forEach((k, v) -> {
                var definitionBuilder = rootBeanDefinition(PooledPBEStringEncryptor.class)
                        .addPropertyValue("algorithm", ALGORITHM)
                        .addPropertyValue("ivGenerator", new RandomIvGenerator())
                        .addPropertyValue("password", v.getKey())
                        .addPropertyValue("poolSize", encryptorPoolSize);

                context.registerBeanDefinition(k, definitionBuilder.getBeanDefinition());
            });
        }
    }

    @Bean
    public EncryptionService encryptionService(KafkaTopicsProperties kafkaTopicsProperties,
                                               Map encryptors) {
        return new EncryptionService(encryptionEnabled, kafkaTopicsProperties, keysProperties, encryptors);
    }
}
